205def compile_match_lists(match_lists: List[dict], cluster_mode: bool = False):
206    """
207    Turns a list of matches of either clusters or candidate pairs found
208    during linkage into a single unified structure holding all found matches
209    across all rules passes. E.g. if a single pass of a linkage algorithm
210    uses three rules, hence generates three dictionaries of matches, this
211    function will aggregate the results of those three separate dicts into
212    a single unified and deduplicated dictionary. For consistency during
213    statistical evaluation, the returned dictionary is always indexed by
214    the lower ID of the records in a given pair.
215
216    :param match_lists: A list of the dictionaries obtained during a run
217      of the linkage algorithm, one dictionary per rule used in the run.
218    :param cluster_mode: An optional boolean indicating whether the linkage
219      algorithm was run in cluster mode. Default is False.
220    :return: The aggregated dictionary of unified matches.
221    """
222    matches = {}
223    for matches_from_rule in match_lists:
224        for matches_within_blocks in matches_from_rule.values():
225            for candidate_set in matches_within_blocks:
226                # Always index the aggregate by the lowest valued ID
227                # for statistical consistency and deduplication
228                root_record = min(candidate_set)
229                if root_record not in matches:
230                    matches[root_record] = set()
231
232                # For clustering, need to add all other records in the cluster
233                if cluster_mode:
234                    for clustered_record in candidate_set:
235                        if clustered_record != root_record:
236                            matches[root_record].add(clustered_record)
237                else:
238                    matched_record = max(candidate_set)
239                    matches[root_record].add(matched_record)
240    return matches

245        """

250        :param block_criteria: a dictionary that contains the blocking criteria.

273        self, organized_block_criteria: dict, query: Select
274    ) -> Select:
275        """
276        Generates a query for selecting a block of data from the MPI tables per the
277        block field criteria.  The block field criteria should be a dictionary
278        organized by MPI table name, with the ORM table object, and the blocking
279        criteria.

285
286        """
287        new_query = query
288
289        for table_key, table_info in organized_block_criteria.items():
290            query_criteria = None
291            cte_query = None
292            sub_query = None
293
294            cte_query_table = table_info["table"]
295            query_criteria = self._generate_where_criteria(
296                table_info["criteria"], table_key
297            )

299            if query_criteria is not None and len(query_criteria) > 0:

301                    cte_query = (
302                        select(cte_query_table.c.patient_id.label("patient_id"))
303                        .where(text(" AND ".join(query_criteria)))
304                        .cte(f"{table_key}_cte")
305                    )
306                else:
307                    fk_query_table = copy.deepcopy(cte_query_table)
308                    fk_info = fk_query_table.foreign_keys.pop()
309                    fk_column = fk_info.column
310                    fk_table = fk_info.column.table
311                    sub_query = (
312                        select(cte_query_table)

318                            sub_query,
