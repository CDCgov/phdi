 29    def __init__(self, resource_location: str = None, scope: str = None):
 30        """
 31        Creates a new AzureCredentialManager object.
 32
 33        :param resource_location: The URL or other location of the requested resource.
 34        :param scope: A space-delimited list of scopes to limit access to resource.
 35          Default: `None`
 36        """
 37        self.__resource_location = resource_location
 38        self.__scope = scope
 39        self.__access_token = None
 40

 42            self.__scope = f"{self.resource_location}/.default"

 44    def get_credential_object(self) -> object:
 45        """
 46        Gets an Azure-specific credential object.
 47
 48        :return: An instance of one of the \\*Credential objects from the
 49          `azure.identity` package.
 50        """
 51        return DefaultAzureCredential()
 52
 53    def get_access_token(self, force_refresh: bool = False) -> str:
 54        """
 55        Obtains an access token from the Azure identity provider. Returns the
 56        access token string, refreshed if expired or force_refresh is specified.
 57
 58        :param force_refresh: `True` if a new token should be requested, regardless
 59          of expiration timestamp. `False` otherwise. Default: `False`
 60        :return: An Azure access token.
 61        """
 62        if force_refresh or (self.access_token is None) or self._need_new_token():
 63            creds = self.get_credential_object()
 64            self.__access_token = creds.get_token(self.scope)
 65
 66        return self.access_token.token
 67
 68    def _need_new_token(self) -> bool:
 69        """
 70        Determines whether the token already stored for this object can be reused,
 71        or if it needs to be re-requested. A new token is needed if a token has not
 72        yet been created, or if the current token has expired.
 73
 74        :return: True if a new Azure access token is needed; false otherwise.
 75        """
 76        try:
 77            current_time_utc = datetime.now(timezone.utc).timestamp()
 78            return self.access_token.expires_on < current_time_utc
 79        except AttributeError:
 80            # access_token not set
 81            return True
 82
 83    def get_secret(self, secret_name: str, key_vault_name: str) -> str:
 84        """
 85        Get the value of a secret from an Azure key vault given the names of the vault
 86        and the secret.
 87
 88        :param secret_name: The name of the secret whose value should be retrieved from
 89            the key vault.
 90        :param key_vault_name: The name of the key vault where the secret is stored.
 91        :return: The value of the secret specified by secret_name.
 92        """
 93
 94        vault_url = f"https://{key_vault_name}.vault.azure.net"
 95        secret_client = SecretClient(
 96            vault_url=vault_url, credential=self.get_credential_object()
 97        )
 98        return secret_client.get_secret(secret_name).value

114    def __init__(self, storage_account_url: str, cred_manager: AzureCredentialManager):
115        """
116        Creates a new AzureCloudContainerConnection object.
117
118        :param storage_account_url: The storage account location of the requested
119          resource.
120        :param cred_manager: The credential manager used to authenticate to the
121          FHIR server.
122        """
123        self.__storage_account_url = storage_account_url
124        self.__cred_manager = cred_manager

126    def _get_container_client(self, container_url: str) -> ContainerClient:
127        """
128        Obtains a client connected to an Azure storage container by
129        utilizing the first valid credentials Azure can find. For
130        more information on the order in which the credentials are
131        checked, see the Azure documentation:
132        https://docs.microsoft.com/en-us/azure/developer/python/sdk/authentication-overview#sequence-of-authentication-methods-when-using-defaultazurecredential
133
134        :param container_url: The url at which to access the container.
135        :return: The Azure `ContainerClient`.
136        """
137        creds = self.cred_manager.get_credential_object()
138        return ContainerClient.from_container_url(container_url, credential=creds)

140    def download_object(
141        self,
142        container_name: str,
143        filename: str,
144        encoding: Literal[None, str] = "UTF-8",
145    ) -> Literal[str, bytes]:
146        """
147        Downloads a blob from storage and returns it as a string or bytes.
148
149        :param container_name: The name of the container containing object to download.
150        :param filename: The location of the file within Azure blob storage.
151        :param encoding: The encoding applied to the downloaded content. If set to None
152            then the blob contents will be returned as bytes. Default: UTF-8
153        :return: A character blob as a string from the given container and filename.
154        """
155        container_location = f"{self.storage_account_url}/{container_name}"
156        container_client = self._get_container_client(container_location)
157        blob_client = container_client.get_blob_client(filename)
158
159        downloader = blob_client.download_blob(encoding=encoding)
160
161        blob_contents = downloader.readall()
162
163        return blob_contents
164
165    def upload_object(
166        self,
167        message: Union[str, dict],
168        container_name: str,
169        filename: str,
170    ) -> None:
171        """
172        Uploads the content of a given message to Azure blob storage.
173        The message can be passed either as a raw string or as JSON.
174
175        :param message: The contents of a message, encoded either as a
176          string or a JSON-formatted dictionary.
177        :param container_name: The name of the target container for upload.
178        :param filename: The location of file to upload within Azure blob storage.
179        """
180        container_location = f"{self.storage_account_url}/{container_name}"
181        container_client = self._get_container_client(container_location)
182        blob_client = container_client.get_blob_client(filename)
183
184        if isinstance(message, str):
185            blob_client.upload_blob(bytes(message, "utf-8"), overwrite=True)
186        elif isinstance(message, dict):
187            blob_client.upload_blob(json.dumps(message).encode("utf-8"), overwrite=True)
188
189    def list_containers(self) -> List[str]:
190        """
191        Lists names for this CloudContainerConnection's containers.
192
193        :return: A list of container names.
194        """
195        creds = self.cred_manager.get_credential_object()
196        service_client = BlobServiceClient(
197            account_url=self.storage_account_url, credential=creds
198        )
199        container_properties_generator = service_client.list_containers()
200
201        container_name_list = []
202        for container_propreties in container_properties_generator:
203            container_name_list.append(container_propreties.name)
204
205        return container_name_list
206
207    def list_objects(self, container_name: str, prefix: str = "") -> List[str]:
208        """
209        Lists names for objects within a container.
210
211        :param container_name: The name of the container to look for objects.
212        :param prefix: Filter the objects returned to filenames beginning
213          with this value.
214        :return: A list of names for objects in given container.
215        """
216        container_location = f"{self.storage_account_url}/{container_name}"
217        container_client = self._get_container_client(container_location)
218
219        blob_properties_generator = container_client.list_blobs(name_starts_with=prefix)
220
221        blob_name_list = []
222        for blob_propreties in blob_properties_generator:
223            blob_name_list.append(blob_propreties.name)
224
225        return blob_name_list

227    def blob_exists(self, container_name: str, filename: str) -> bool:
228        """
229        Check if a blob exists within a container given its name and the name of the
230        container.
231
232        :param container_name: The name of the container to look for the blob in.
233        :param filename: The name of the blob to check the existence of.
234        :param prefix: Filter the objects returned to filenames beginning
235          with this value.
236        :return: A boolean of true if the file exists and false if it does not.
237        """
238
239        container_location = f"{self.storage_account_url}/{container_name}"
240        container_client = self._get_container_client(container_location)
241        blob_client = container_client.get_blob_client(filename)
242        return blob_client.exists()
